import numpy as np import wave from js import document, URL, Blobs from pyodide.ffi import to_js from backend.transcribeaudio import transcribe_audio, meeting_minutes, save_as_docx, save_as_pdf def save_wav_file(blob): array_buffer = blob.arrayBuffer().to_py() audio_data = np.frombuffer(array_buffer, dtype=np.int16) with wave.open('recorded_audio.wav', 'wb') as wf: wf.setnchannels(1) # Assuming mono channel wf.setsampwidth(2) # Assuming 16-bit samples wf.setframerate(44100) # Assuming 44100 Hz sample rate wf.writeframes(audio_data) result = transcribe_audio("recorded_audio.wav") minutes = meeting_minutes(result) filename = document.getElementById("audio-file-input").value format_choice = document.querySelector('input[name="audio-doc-format"]:checked').value if format_choice == "docx": save_as_docx(minutes, filename + ".docx") elif format_choice == "pdf": save_as_pdf(minutes, filename + ".pdf") document.getElementById("transcription-output").innerText = "Document saved as " + filename + "." + format_choice window.save_wav_file = save_wav_file
